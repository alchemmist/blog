{{- /* Картинка с оптимизацией (#1). Принимает dict: src (обяз.), alt, title, class, id, loading (по умолч. lazy), maxw (ширина показа в px — из obsidian-синтаксиса ![|400]). Локальные растры (png/jpg/webp) конвертит в WebP q80 (с потолком 1600px) и отдаёт одним файлом + width/height (против CLS). Ширину показа задаёт CSS, а не srcset — иначе medium-zoom не увеличивает картинку при зуме (см. ниже). GIF/SVG/удалённые — отдаёт как есть. Класс content-img гарантирует height:auto + max-width:100% (против растягивания). */ -}} {{- $src := .src -}} {{- $alt := .alt | default "" -}} {{- $title := .title -}} {{- $class := trim (printf "content-img %s" (.class | default "")) " " -}} {{- $id := .id -}} {{- $loading := .loading | default "lazy" -}} {{- $maxw := int (.maxw | default 0) -}} {{- $style := cond (gt $maxw 0) (printf "width:%dpx" $maxw) "" -}} {{- $ext := lower (strings.TrimPrefix "." (path.Ext $src)) -}} {{- $isLocal := and (hasPrefix $src "/") (not (hasPrefix $src "//")) -}} {{- $raster := in (slice "png" "jpg" "jpeg" "webp") $ext -}} {{- $res := "" -}} {{- if and $isLocal $raster -}} {{- $path := strings.TrimPrefix "/" $src -}} {{- with (resources.Get $path) -}} {{- $res = . -}} {{- else -}} {{- with (resources.Get (replace $path "%20" " ")) -}}{{- $res = . -}}{{- end -}} {{- end -}} {{- end -}} {{- if $res -}} {{- /* Один оптимизированный WebP, БЕЗ srcset/sizes. Ширину показа задаём через CSS (style width / max-width), браузер сам масштабирует. Почему не srcset: при srcset+sizes браузер отдаёт naturalWidth, скорректированный по density до ширины показа, а medium-zoom считает по нему масштаб зума — из-за этого зум переставал увеличивать картинку (регрессия #1). Плюс density «прилипает» к HD-клону, так что data-zoom-src это не лечит. Отдаём полноразмерный кадр (с потолком 1600px) — зум открывает его в исходном качестве, а payload всё равно мал: WebP q80 + даунскейл. */ -}} {{- $cap := 1600 -}} {{- $targetW := cond (gt $res.Width $cap) $cap $res.Width -}} {{- $img := $res.Resize (printf "%dx webp q80" $targetW) -}} {{- else -}} {{- end -}}

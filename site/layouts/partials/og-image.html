{{/* Возвращает URL OG-картинки. Для контентных страниц генерит карточку 1200x630: слева раздел + заголовок (центрированы по своей зоне), справа — иллюстрация со страницы, если она есть. Заголовок центрируется, чтобы квадратный кроп мессенджеров (MAX и т.п.) его не срезал. Иначе — preview/Params.image. (#3) */}} {{/* Возвращаем КОРНЕ-ОТНОСИТЕЛЬНЫЙ URL (/og/... или /assets/...). Абсолютным его делает вызывающий (absURL) для og:image/twitter/JSON-LD, а Pagefind получает относительный — чтобы превью в поиске грузилось и локально, и на проде, а не всегда с продакшена (иначе не задеплоенные карточки дают 404). (#13) */}} {{ $default := "/assets/preview.png" }} {{ with .Params.image }}{{ $default = . }}{{ end }} {{ $result := $default }} {{ $sections := slice "articles" "essays" "moss" "teach" "poetry" }} {{ if and .IsPage (in $sections .Section) (not .Params.image) }} {{ $bg := resources.Get "og/bg.png" }} {{ $font := resources.Get "og/font-bold.ttf" }} {{ if and $bg $font }} {{ $img := $bg }} {{/* Иллюстрация — первая картинка из РЕАЛЬНОЙ разметки контента: markdown ![..](/images/..) или . Сканировать весь RawContent нельзя — путь, упомянутый в прозе или код-блоке, дал бы ложное совпадение. */}} {{ $ill := "" }} {{ range (findRE `(?:!\[[^\]]*\]\(|]*\bsrc=["']?)\s*/images/[^)\s"'>]+\.(?i:png|jpe?g|webp)` .RawContent) }} {{ if eq $ill "" }} {{ with (index (findRE `/images/[^)\s"'>]+\.(?i:png|jpe?g|webp)` .) 0) }} {{ $p := strings.TrimPrefix "/" (replace . "%20" " ") }} {{ with (resources.Get $p) }}{{ $ill = . }}{{ end }} {{ end }} {{ end }} {{ end }} {{ $hasIll := ne $ill "" }} {{/* Иллюстрация — целиком (с тенью и скруглением) вписывается в правую половину с отступами и центрируется, чтобы тень не срезалась по краям */}} {{ if $hasIll }} {{ $fit := $ill.Fit "500x520" }} {{ $ox := add 600 (div (sub 600 $fit.Width) 2) }} {{ $oy := div (sub 630 $fit.Height) 2 }} {{ $img = $img.Filter (images.Overlay $fit $ox $oy) }} {{ end }} {{/* Зона текста: левая половина (если есть илл.) или вся карточка */}} {{ $cx := cond $hasIll 300 600 }} {{ $titleSize := 60 }} {{ $sectionSize := 28 }} {{ $lineH := 76 }} {{ $gap := 22 }} {{ $max := cond $hasIll 14 28 }} {{/* Перенос заголовка по словам (кириллица учтена) */}} {{ $lines := slice }} {{ $cur := "" }} {{ range (split .Title " ") }} {{ $try := cond (eq $cur "") . (printf "%s %s" $cur .) }} {{ if le (strings.RuneCount $try) $max }} {{ $cur = $try }} {{ else }} {{ if ne $cur "" }}{{ $lines = $lines | append $cur }}{{ end }} {{ $cur = . }} {{ end }} {{ end }} {{ if ne $cur "" }}{{ $lines = $lines | append $cur }}{{ end }} {{ $lines = first 4 $lines }} {{/* Вертикальное центрирование блока (раздел + зазор + строки) */}} {{ $blockH := add (add $sectionSize $gap) (mul (len $lines) $lineH) }} {{ $startY := div (sub 630 $blockH) 2 }} {{ if lt $startY 48 }}{{ $startY = 48 }}{{ end }} {{/* Раздел — по центру зоны */}} {{ $sec := upper .Section }} {{ $secW := int (mul (mul (strings.RuneCount $sec) $sectionSize) 0.62) }} {{ $secX := sub $cx (div $secW 2) }} {{ if lt $secX 24 }}{{ $secX = 24 }}{{ end }} {{ $img = $img.Filter (images.Text $sec (dict "size" $sectionSize "color" "#8a8a8a" "x" $secX "y" $startY "font" $font)) }} {{/* Заголовок — каждая строка центрируется по cx */}} {{ range $i, $line := $lines }} {{ $w := int (mul (mul (strings.RuneCount $line) $titleSize) 0.56) }} {{ $x := sub $cx (div $w 2) }} {{ if lt $x 24 }}{{ $x = 24 }}{{ end }} {{ $y := add (add $startY $sectionSize) (add $gap (mul $i $lineH)) }} {{ $img = $img.Filter (images.Text $line (dict "size" $titleSize "color" "#1a1a1a" "x" $x "y" $y "font" $font)) }} {{ end }} {{ $result = $img.RelPermalink }} {{ end }} {{ end }} {{ return $result }}
